1 define(["FsioVersion"], function(ver) {
  2 
  3     /**
  4      * @fileOverview
  5      *
  6      * Abstract base class for all FSIO namespaces.
  7      */
  8 
  9     /**
 10      * Base class constructor. Shouldn't be instantiated directly
 11      * but via subclassing.
 12      *
 13      * @constructor
 14      * @name FsioBase
 15      * @protected
 16      */
 17     function FsioBase() {}
 18 
 19     FsioBase.prototype.init = function(host, name, version, dac, headers) {
 20         this.host = host;
 21         this.dac = dac;
 22         this.prefix = "/" + name + "/" + version + "/";
 23         this.global_headers = headers;
 24     };
 25 
 26     FsioBase.prototype.request = function(
 27         type, path, complete, data, headers, withCredentials) {
 28         headers = headers ? headers : {};
 29         if(this.dac)
 30             headers["x-dac"] = this.dac;
 31         if(this.global_headers) {
 32             // append global headers
 33             for(var p in this.global_headers)
 34                 headers[p] = this.global_headers[p];
 35         }
 36         if(type == "GET" && data) {
 37             path += "?" + $.param(data);
 38             data = null;
 39         }
 40         var settings = {
 41             url: this.host + this.prefix + path,
 42             type: type,
 43             data: data, // query params with GET
 44             headers: headers,
 45             complete: [function(jqXHR, textStatus) {
 46                 console.log("in <==\nstatus: " +
 47                               jqXHR.status +
 48                               "\nheaders: " +
 49                               jqXHR.getAllResponseHeaders());
 50             }, complete],
 51             beforeSend: function(jqXHR, settings) {
 52                 console.log("out ==> ", settings);
 53             },
 54             cache: false,
 55             contentType: false,
 56             processData: false,
 57             xhrFields: {
 58                 withCredentials: withCredentials ? withCredentials : false
 59             }
 60         };
 61         $.ajax(settings);
 62     };
 63 
 64     FsioBase.prototype.requestWithTicket =
 65         function(ticket, type, path, complete, data, headers) {
 66             headers = headers ? headers : {};
 67             headers["x-application-ticket"] = ticket;
 68             this.request(type, path, complete, data, headers);
 69         };
 70 
 71     FsioBase.prototype.requestWithUploadToken =
 72         function(token, type, path, complete, data, headers) {
 73             headers = headers ? headers : {};
 74             headers["x-upload-token"] = token;
 75             headers["content-type"] = headers["content-type"] ?
 76                 headers["content-type"] : "application/octet-stream";
 77             this.request(type, path, complete, data, headers);
 78         };
 79 
 80     // Return the construtor function
 81     return FsioBase;
 82 });
