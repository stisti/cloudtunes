1 define(["FsioBase"], function(FsioBase) {
  2 
  3     /**
  4      * @fileOverview
  5      *
  6      * FSIO Data API.
  7      */
  8 
  9     FsioData.prototype = new FsioBase(); // inherit
 10     FsioData.prototype.constructor = FsioData;
 11 
 12     /**
 13      * @class
 14      * @name FsioData
 15      * @augments FsioBase
 16      * @see Fsio#data
 17      */
 18     function FsioData(host, version, dac, headers) {
 19         this.init(host, "data", version, dac, headers);
 20     }
 21 
 22     /**
 23      * Upload data.
 24      *
 25      * @param {string} token       FSIO upload token.
 26      * @param {string} object_name The name of the object being uploaded.
 27      * @param {string|Blob} data   Data to upload.
 28      * @param {function(jqXHR,textStatus)} complete
 29      * Complete callback.
 30      * @param {object} [params]    Query parameters.
 31      * @param {object} [headers]   Extra headers.
 32      */
 33     FsioData.prototype.upload = function(
 34         token, object_name, data, complete, params, headers) {
 35         var path = "upload/" + object_name;
 36         if(params) {
 37             // query params for PUT
 38             path += "?" + $.param(params);
 39         }
 40         this.requestWithUploadToken(token, "PUT", path, complete, data,
 41                                     headers);
 42     };
 43 
 44     /**
 45      * Initialize partial upload. Upload ID is returned with 'upload-id'
 46      * response header.
 47      *
 48      * @param {string} token       FSIO upload token.
 49      * @param {string} object_name The name of the object being uploaded.
 50      * @param {function(jqXHR,textStatus)} complete
 51      * Complete callback.
 52      * @param {object} [params]    Query parameters.
 53      * @param {object} [headers]   Extra headers.
 54      */
 55     FsioData.prototype.partialUploadInit = function(
 56         token, object_name, complete, params, headers) {
 57         var path = "upload_partial/" + object_name;
 58         if(params) {
 59             // query params for POST
 60             path += "?" + $.param(params);
 61         }
 62         this.requestWithUploadToken(token, "POST", path, complete, null,
 63                                     headers);
 64     };
 65 
 66     /**
 67      * Query partial upload state. Bytes uploaded so far are returned with
 68      * 'bytes' key in response json.
 69      * Applications should normally use {@link #uploadPartially} to upload
 70      * data partially.
 71      *
 72      * @param {string} token       FSIO upload token.
 73      * @param {integer} upload_id  Upload ID received from
 74      * {@link #partialUploadInit}.
 75      * @param {function(jqXHR,textStatus)} complete
 76      * Complete callback.
 77      * @param {object} [params]    Query parameters.
 78      * @param {object} [headers]   Extra headers.
 79      */
 80     FsioData.prototype.partialUploadQuery = function(
 81         token, upload_id, complete, params, headers) {
 82         this.requestWithUploadToken(token, "GET",
 83                                     "upload_partial/" + upload_id,
 84                                     complete, params, headers);
 85     };
 86 
 87     /**
 88      * Upload next part. Use 'content-range' header to specify the part being
 89      * uploaded, e.g. {'content-range':'bytes 0-10000/100000'}.
 90      * Applications should normally use {@link #uploadPartially} to upload
 91      * data partially.
 92      *
 93      * @param {string} token       FSIO upload token.
 94      * @param {integer} upload_id  Upload ID received from
 95      * {@link #partialUploadInit}.
 96      * @param {string|Blob} data   Data to upload, size must match the
 97      * 'content-range' header.
 98      * @param {function(jqXHR,textStatus)} complete
 99      * Complete callback.
100      * @param {object} [params]    Query parameters.
101      * @param {object} [headers]   Extra headers.
102      */
103     FsioData.prototype.partialUpload = function(
104         token, upload_id, data, complete, params, headers) {
105         var path = "upload_partial/" + upload_id;
106         if(params) {
107             // query params for PUT
108             path += "?" + $.param(params);
109         }
110         this.requestWithUploadToken(token, "PUT", path, complete, data,
111                                     headers);
112     };
113 
114     /**
115      * Upload data partially, uses internally {@link #partialUploadQuery}
116      * and {@link #partialUpload}.
117      *
118      * @param {string} token       FSIO upload token.
119      * @param {integer} upload_id  Upload ID received from
120      * {@link #partialUploadInit}.
121      * @param {string|Blob} data   Data to upload.
122      * @param {integer} max_chunk_size Max size of upload chunk in bytes.
123      * @param {function(jqXHR,textStatus)} complete
124      * Complete callback.
125      * @param {object} [params]    Query parameters.
126      * @param {object} [headers]   Extra headers.
127      */
128     FsioData.prototype.uploadPartially = function(
129         token, upload_id, data, max_chunk_size, complete, params, headers) {
130         var hdrs = headers ? headers : {};
131         var data_len = 0;
132         if(data instanceof Blob) {
133             data.slice = data.slice || data.webkitSlice || data.mozSlice;
134             data_len = data.size;
135         } else {
136             // String
137             data_len = data.length;
138         }
139         // loop until partial upload done or error occured
140         (function loop(self) {
141             // query first partial upload state
142             self.partialUploadQuery(token, upload_id, function(
143                 jqXHR, textStatus) {
144                 if(jqXHR.status != 200) {
145                     complete(jqXHR, textStatus);
146                 } else {
147                     var start_pos = JSON.parse(jqXHR.responseText).bytes;
148                     var chunk_size = Math.min(max_chunk_size,
149                                               data_len - start_pos);
150                     var end_pos = start_pos + chunk_size;
151                     var range = "bytes " + start_pos + "-" + (end_pos-1) + "/" +
152                         data_len;
153                     console.log("Uploading with id " + upload_id +
154                                   ", range: " + range);
155                     hdrs["content-range"] = range;
156                     // upload next part
157                     var part = data instanceof Blob ?
158                         data.slice(start_pos, end_pos, data.type) :
159                         data.substring(start_pos, end_pos);
160                     self.partialUpload(token, upload_id, part, function(jqXHR, textStatus) {
161                         // done or error?
162                         if(jqXHR.getResponseHeader("content-id") ||
163                            jqXHR.status != 204) {
164                             complete(jqXHR, textStatus);
165                         } else {
166                             // continue with upload
167                             loop(self);
168                         }
169                     }, params, hdrs);
170                 }
171             });
172         })(this);
173     };
174 
175     /**
176      * Download data. To download the specified byte range of the file use
177      * 'range' header, e.g. {'range': 'bytes=0-1000'}.
178      *
179      * @param {string} token       FSIO ticket.
180      * @param {string} object_name The name of the object being downloaded.
181      * @param {function(jqXHR,textStatus)} complete
182      * Complete callback.
183      * @param {object} [params]    Query parameters.
184      * @param {object} [headers]   Extra headers.
185      */
186     FsioData.prototype.download = function(
187         ticket, object_name, complete, params, headers) {
188         params = params ? params : {};
189         params.ticket = ticket;
190         // do simple CORS request
191         this.request("GET", "files/" + object_name, complete, params, headers);
192     };
193 
194     // Return the constructor function
195     return FsioData;
196 });
