 12     /**

 18     function FsioContent(host, version, dac, headers) {

 22     /**

 28      * @param {object} [params]    Query parameters.

 30      */
 31     FsioContent.prototype.getUserAccountInfo = function(
 32         ticket, complete, params, headers) {
 33         this.requestWithTicket(ticket, "GET", "account", complete,
 34                                params, headers);
 35     };

 53     /**
 54      * Deletes a file or folder matching the path.
 55      *
 56      * @param {string} ticket      FSIO ticket.
 57      * @param {string} path        Full path of the file to delete.
 58      * @param {function(jqXHR,textStatus)} complete
 59      * Complete callback.
 60      * @param {object} [params]    Query parameters.
 61      * @param {object} [headers]   Extra headers.
 62      */
 63     FsioContent.prototype.deleteFile = function(
 64         ticket, path, complete, params, headers) {
 65         headers = headers ? headers : {};
 66         headers["content-type"] = "";
 67         var deletepath = "files/" + path;
 68         if(params) {
 69             // query params for DELETE
 70             deletepath += "?" + $.param(params);
 71         }
 72         this.requestWithTicket(ticket, "DELETE", deletepath, complete,
 73                                null, headers);
 74     };
 75

 99     /**
100      * Move or rename a file or a folder.
101      *
102      * @param {string} ticket      FSIO ticket.
103      * @param {string} from        Full path of the file or folder to move.
104      * @param {string} to          Full path of the new file or folder.
105      * @param {function(jqXHR,textStatus)} complete
106      * Complete callback.
107      * @param {object} [params]    Query parameters.
108      * @param {object} [headers]   Extra headers.
109      */
110     FsioContent.prototype.moveFile = function(
111         ticket, from, to, complete, params, headers) {
112         headers = headers ? headers : {};
113         headers["content-type"] = "application/json; charset=UTF-8";
114         // json body for request
115         var data = JSON.stringify({
116             "Key": "/files/" + to
117         });
118         var path = "files/" + from;
119         if(params) {
120             // query params for POST
121             path += "?" + $.param(params);
122         }
123         this.requestWithTicket(ticket, "POST", path, complete,
124                                data, headers);
125     };

205     /**
206      * Set description and/or user metadata for a file.
207      *
208      * @param {string} ticket      FSIO ticket.
209      * @param {string} path        Full path of the file.
210      * @param {string} description Description of the entry.
211      * @param {object} metadata    Key-value pairs of user-specified metadata.
212      * @param {function(jqXHR,textStatus)} complete
213      * Complete callback.
214      * @param {object} [params]    Query parameters.
215      * @param {object} [headers]   Extra headers.
216      */
217     FsioContent.prototype.setFileMetadata = function(
218         ticket, path, description, metadata, complete, params, headers) {
219         headers = headers ? headers : {};
220         headers["content-type"] = "application/json; charset=UTF-8";
221         var json = {};
222         if(null !== description)
223             json.Description = description;
224         if(null !== metadata)
225             json.UserMetadata = metadata;
226         var data = JSON.stringify(json);
227         var setpath = "files/" + path;
228         if(params) {
229             // query params for POST
230             setpath += "?" + $.param(params);
231         }
232         this.requestWithTicket(ticket, "POST", setpath, complete,
233                                data, headers);
234     };
