  9     /**
 10      * Create an instance of CAN client.
 11      *
 12      * @constructor
 13      * @name Can
 14      * @this {Can}
 15      * @param {object} config Can config.
 16      * @param {string} api_key Developer API key.
 17      * @param {string} api_key_secret Developer API key secret.
 18      * @throws {InvalidParamError} If missing parameter.
 19      * @throws {InvalidConfigError} If invalid config.
 20      */
 21     function Can(config,
 22                  api_key,
 23                  api_key_secret /* not used currently */) {
 24         if(!config)
 25             throw { name:"InvalidParamError", message:"Missing parameter" };
 26         this._checkConfig(config);
 27         this.config = config;
 28         // XXX: API key should be hashed with given secret.
 29         // But we don't have any crypto in js...
 30         this.api_key = api_key;
 31         this.can_login_url = null;
 32         this.uuid = null;
 33     }

 35     /** Get version. */
 36     Can.prototype.version = function() /** string */ {

 40     Can.prototype._checkConfig = function(config) {
 41         if(!config.ONEID_HOST)
 42             throw { name:"InvalidConfigError",
 43                     message:"Missing ONEID_HOST" };
 44         if(!config.SEBE_HOST)
 45             throw { name:"InvalidConfigError",
 46                     message:"Missing SEBE_HOST" };
 47         if(!config.FSIO_PSK)
 48             throw { name:"InvalidConfigError",
 49                     message:"Missing FSIO_PSK" };
 50         if(!config.FSIO_TICKET_HOST)
 51             throw { name:"InvalidConfigError",
 52                     message:"Missing FSIO_TICKET_HOST" };
 53         if(!config.FSIO_TICKET_DAC)
 54             throw { name:"InvalidConfigError",
 55                     message:"Missing FSIO_TICKET_DAC" };
 56         if(!config.FSIO_API_HOST)
 57             throw { name:"InvalidConfigError",
 58                     message:"Missing FSIO_API_HOST" };
 59         if(!config.FSIO_API_DAC)
 60             throw { name:"InvalidConfigError",
 61                     message:"Missing FSIO_API_DAC" };
 62         if(!config.FSIO_DATA_HOST)
 63             throw { name:"InvalidConfigError",
 64                     message:"Missing FSIO_DATA_HOST" };
 65         if(!config.FSIO_DATA_DAC)
 66             throw { name:"InvalidConfigError",
 67                     message:"Missing FSIO_DATA_DAC" };
 68         if(!config.UEB_HOST)
 69             throw { name:"InvalidConfigError",
 70                     message:"Missing UEB_HOST" };
 71         if(!config.CP_LINK_HOST)
 72             throw { name:"InvalidConfigError",
 73                     message:"Missing CP_LINK_HOST" };
 74         if(!config.APPLICATION_FAMILY_ID)
 75             throw { name:"InvalidConfigError",
 76                     message:"Missing APPLICATION_FAMILY_ID" };
 77         if(!config.ONEID_APP_ID)
 78             throw { name:"InvalidConfigError",
 79                     message:"Missing ONEID_APP_ID" };
 80         if(!config.OPERATOR_ID)
 81             throw { name:"InvalidConfigError",
 82                     message:"Missing OPERATOR_ID" };
 83     };

 85     Can.prototype._createHeaders = function() {
 86         var headers = {};
 87         if(this.api_key)
 88             headers["x-apikey"] = this.api_key;
 89         return headers;
 90     };

 92     Can.prototype._request = function(type, url, data, headers, complete) {
 93         var settings = {
 94             url: url,
 95             type: type,
 96             data: data, // query params with GET
 97             headers: headers,
 98             complete: [function(jqXHR, textStatus) {
 99                 console.log("in <==\nstatus: " +
100                               jqXHR.status +
101                               "\nheaders: " +
102                               jqXHR.getAllResponseHeaders());
103             }, complete],
104             beforeSend: function(jqXHR, settings) {
105                 console.log("out ==> ", settings);
106             },
107             xhrFields: {
108                 withCredentials: true
109             }
110         };
111         $.ajax(settings);
112     };

114     // get UUID from OneID
115     Can.prototype._getUUID = function(complete) {
116         var url = this.config.ONEID_HOST +
117             "/OneID/user/2_0_0/scim/v1/Users/current";
118         var headers = this._createHeaders();
119         this._request("GET", url, null, headers, function(
120             jqXHR, textStatus) {
121             if(jqXHR.status != 200) {
122                 console.log("Failed to get UUID from OneID: " + jqXHR.status);
123                 console.log(jqXHR.responseText);
124             }
125             complete(jqXHR, textStatus);
126         });
127     };

129     // get SEBE ticket

132             "/ticketing/services/spi/tickets/sync";
133         var data = {
134             operatorId: this.config.OPERATOR_ID,
135             version: 55
136         };
137         var headers = this._createHeaders();
138         this._request("GET", url, data, headers, function(
139             jqXHR, textStatus) {
140             if(jqXHR.status != 200) {
141                 console.log("Failed to get SEBE ticket: " + jqXHR.status);
142                 console.log(jqXHR.responseText);
143                 var error = $(jqXHR.responseText).find("code").text();
144                 // not registered yet?
145                 if(error == 6030) {
146                     console.log("User is not registered yet");
147                     // XXX: SEBE registration API requires OneID token
148                     // in xml body. But we cannot access 'HttpOnly' cookie
149                     // from JavaScript.
150                 }
151             }
152             complete(jqXHR, textStatus);
153         });

156     /**
157      * Get URL for OAuth login. After successful login 'authenticationToken' is
158      * returned via cookies in response. Use {@link #login} to authenticate
159      * user.
160      *
161      * @param {string} post_auth_url Url to load after successful login.
162      * Must be url encoded.
163      * @param {function(status,url)} complete
164      * Complete callback with http status code and url.
165      */
166     Can.prototype.getLoginUrl = function(post_auth_url, complete) {
167         if(this.can_login_url) {
168             complete(200, this.can_login_url);
169         } else {
170             var url = this.config.ONEID_HOST +
171                 "/OneID/sso/2_0_0/discovery/providers";
172             var data = {
173                 app_id: this.config.ONEID_APP_ID,
174                 operator_key: this.config.OPERATOR_ID
175             };
176             this._request("GET", url, data, this._createHeaders(), function(
177                 jqXHR, textStatus) {
178                 if(jqXHR.status != 200) {
179                     console.log("Failed to get login url: " + jqXHR.status);
180                     console.log(jqXHR.responseText);
181                     complete(jqXHR.status);
182                 } else {
183                     var json = JSON.parse(jqXHR.responseText);
184                     var urlbase =
185                         json.providers[0].methods[0].authorize;
186                     data.redirect_url = post_auth_url;
187                     this.can_login_url = urlbase + "?" + $.param(data);
188                     complete(jqXHR.status, this.can_login_url);
189                 }
190             });
191         }
192     };

194     /**
195      * Login after cookies have been populated with authentication token.
196      *
197      * @param {function(status)} complete
198      * Complete callback with http status code.
199      */
200     Can.prototype.login = function(complete) {
201         (function(self) {
202             // get first UUID from OneID (we need it for UEB)
203             self._getUUID(function(jqXHR, textStatus) {
204                 if(jqXHR.status != 200) {
205                     complete(jqXHR.status);
206                 } else {
207                     // store UUID
208                     self.uuid = JSON.parse(jqXHR.responseText).id;
209                     console.log("Parsed UUID: " + self.uuid);
210                     // get SEBE ticket
211                     self._getSEBETicket(function(
212                         jqXHR, textStatus) {
213                         complete(jqXHR.status);
214                     });
215                 }
216             });
217         })(this);
218     };

225     /**
226      * Create FSIO client after successful login.
227      *
228      * @throws {NotInitializedError} If not logged in.
229      * @returns {object} FSIO client instance.
230      */
231     Can.prototype.createFsioClient = function() {
232         if(!this.uuid)
233             throw { name:"NotInitializedError", message:"No UUID" };
234         var global_headers = {};
235         if(this.api_key)
236             global_headers["x-apikey"] = this.api_key;
237         return new Fsio(this.uuid,
238                         this.config.FSIO_PSK,
239                         this.config.FSIO_TICKET_HOST,
240                         this.config.FSIO_TICKET_DAC,
241                         this.config.FSIO_API_HOST,
242                         this.config.FSIO_API_DAC,
243                         this.config.FSIO_DATA_HOST,
244                         this.config.FSIO_DATA_DAC,
245                         this.config.UEB_HOST,
246                         global_headers);
247     };

249     /**
250      * Create user ticket for FSIO API. When ticket has expired FSIO API calls
251      * return http status 401 with 128 as FSIO error code.
252      *
253      * @param {object} fsio FSIO client.
254      * @param {function(status,ticket)} complete
255      * Complete callback with http status code and ticket.
256      * @param {integer} [ttl] TTL for the ticket in seconds (1800-3600).
257      */
258     Can.prototype.createFsioUserTicket = function(fsio, complete, ttl) {
259         var params = {};
260         if(ttl)
261             params.ttl = ttl;
262         (function() {
263             fsio.ticket.createWithCookies(function(jqXHR, textStatus) {
264                 var ticket = null;
265                 if(jqXHR.status == 200) {
266                     ticket = JSON.parse(jqXHR.responseText).Ticket;
267                 } else {
268                     console.log("Failed to get FSIO ticket: " + jqXHR.status);
269                     console.log(jqXHR.responseText);
270                 }
271                 complete(jqXHR.status, ticket);
272             }, params);
273         })();
274     };
