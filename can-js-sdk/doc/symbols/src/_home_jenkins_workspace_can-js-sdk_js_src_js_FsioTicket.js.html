1 define(["FsioBase"], function(FsioBase) {
  2 
  3     /**
  4      * @fileOverview
  5      *
  6      * FSIO Ticket API.
  7      */
  8 
  9     FsioTicket.prototype = new FsioBase(); // inherit
 10     FsioTicket.prototype.constructor = FsioTicket;
 11 
 12     /**
 13      * @class
 14      * @name FsioTicket
 15      * @augments FsioBase
 16      * @see Fsio#ticket
 17      */
 18     function FsioTicket(host, version, dac, headers) {
 19         this.init(host, "ticket", version, dac, headers);
 20     }
 21 
 22     /**
 23      * Create ticket with OneID token and SEBE ticket. Ticket is returned with
 24      * 'Ticket' key in response json.
 25      *
 26      * @param {string} oneid_token OneID token.
 27      * @param {string} sebe_ticket SEBE ticket.
 28      * @param {function(jqXHR,textStatus)} complete
 29      * Complete callback.
 30      * @param {object} [params]    Query parameters.
 31      * @param {object} [headers]   Extra headers.
 32      */
 33     FsioTicket.prototype.createWithAuth =
 34         function(oneid_token, sebe_ticket, complete, params, headers) {
 35             headers = headers ? headers : {};
 36             headers["content-type"] = "application/json; charset=UTF-8";
 37             // json body for POST
 38             var data = JSON.stringify({
 39                 "AuthenticationToken": oneid_token,
 40                 "AuthorizationTicket": sebe_ticket
 41             });
 42             // query params for POST
 43             var path = params ? "?" + $.param(params) : "";
 44             this.request("POST", path, complete, data, headers);
 45         };
 46 
 47     /**
 48      * Create ticket with OneID token and SEBE ticket in browser's cookies.
 49      * Ticket is returned with 'Ticket' key in response json.
 50      *
 51      * @param {function(jqXHR,textStatus)} complete
 52      * Complete callback.
 53      * @param {object} [params]    Query parameters.
 54      * @param {object} [headers]   Extra headers.
 55      */
 56     FsioTicket.prototype.createWithCookies =
 57         function(complete, params, headers) {
 58             headers = headers ? headers : {};
 59             headers["content-type"] = "";
 60             var path = "user";
 61             if(params) {
 62                 // query params for POST
 63                 path += "?" + $.param(params);
 64             }
 65             this.request("POST", path, complete, null, headers, true);
 66         };
 67 
 68     /**
 69      * Create upload token. Token is returned with 'Token' key in response json.
 70      *
 71      * @param {string} ticket      FSIO ticket.
 72      * @param {function(jqXHR,textStatus)} complete
 73      * Complete callback.
 74      * @param {object} [params]    Query parameters.
 75      * @param {object} [headers]   Extra headers.
 76      */
 77     FsioTicket.prototype.createUploadToken = function(
 78         ticket, complete, params, headers) {
 79         headers = headers ? headers : {};
 80         headers["content-type"] = "";
 81         var path = "upload";
 82         if(params) {
 83             // query params for POST
 84             path += "?" + $.param(params);
 85         }
 86         this.requestWithTicket(ticket, "POST", path, complete, null, headers);
 87     };
 88 
 89     /**
 90      * Delete ticket.
 91      *
 92      * @param {string} ticket      FSIO ticket to delete.
 93      * @param {function(jqXHR,textStatus)} complete
 94      * Complete callback.
 95      * @param {object} [params]    Query parameters.
 96      * @param {object} [headers]   Extra headers.
 97      */
 98     FsioTicket.prototype.deleteTicket =
 99         function(ticket, complete, params, headers) {
100             headers = headers ? headers : {};
101             headers["content-type"] = "";
102             // query params for DELETE
103             var path = params ? "?" + $.param(params) : "";
104             this.requestWithTicket(ticket, "DELETE", path, complete, null,
105                                    headers);
106         };
107 
108     // Return the constructor function
109     return FsioTicket;
110 });
