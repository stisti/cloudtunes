 10     var FSIO_API_VERSION = "1_0_0";
 11 
 12     /**
 13      * Creates an instance of Fsio client.
 14      *
 15      * @constructor
 16      * @name Fsio
 17      * @this {Fsio}
 18      * @param {string} uid         User ID.
 19      * @param {string} psk         Pre-shared key (not used currently).
 20      * @param {string} ticket_host FSIO Ticket API host.
 21      * @param {string} ticket_dac  FSIO Ticket API DAC.
 22      * @param {string} api_host    FSIO API host.
 23      * @param {string} api_dac     FSIO API DAC.
 24      * @param {string} data_host   FSIO Data host.
 25      * @param {string} data_dac    FSIO Data API DAC.
 26      * @param {string} ueb_host    FSIO UEB host.
 27      * @param {object} [global_headers] Global headers. Used with all requests.
 28      * @property {FsioTicket} ticket FSIO Ticket API.
 29      * @property {FsioContent} content FSIO Content API.
 30      * @property {FsioData} data FSIO Data API.
 31      * @property {FsioSearch} search FSIO Search API.
 32      * @property {FsioUeb} ueb FSIO Notification API.
 33      */
 34     function Fsio(uid,
 35                   psk, // not used
 36                   ticket_host, ticket_dac,
 37                   api_host, api_dac,
 38                   data_host, data_dac,
 39                   ueb_host,
 40                   global_headers) {
 41         this.ticket = new FsioTicket(ticket_host, FSIO_API_VERSION, ticket_dac,
 42                                      global_headers);
 43         this.content = new FsioContent(api_host, FSIO_API_VERSION, api_dac,
 44                                        global_headers);
 45         // XXX: Data API is not routed through API GW
 46         this.data = new FsioData(data_host, FSIO_API_VERSION);
 47         this.search = new FsioSearch(api_host, FSIO_API_VERSION, api_dac,
 48                                      global_headers);
 49         this.ueb = new FsioUeb(ueb_host, FSIO_API_VERSION, uid, global_headers);
 50     }
 51 
 52     /** Get version. */
 53     Fsio.prototype.version = function() /** string */ {
 54         return ver.version;
 55     };

 57     function _getWorkerStatus(json, key) {
 58         var worker = json[key];
 59         if(worker)
 60             return worker.status;
 61     }
 62 
 63     /**
 64      * Wait for the workers to finish processing uploaded file.
 65      * Only for testing purposes.
 66      *
 67      * @param {string} ticket         FSIO ticket.
 68      * @param {string} object_name    Name of the uploaded object.
 69      * @param {string[]} workers      Array of workers, e.g.
 70      * ["AV","FileTypeWorkerStatus","MetadataWorkerStatus"]
 71      * @param {integer} timeout_secs  Timeout in seconds.
 72      * @param {function(statusText)} complete
 73      * Complete callback, called with 'success', 'error' or 'timedout'.
 74      */
 75     Fsio.prototype.waitForWorkers = function(
 76         ticket, object_name, workers, timeout_secs, complete) {
 77         var then = new Date().getTime() + (timeout_secs * 1000);
 78         var params = {
 79             key: object_name,
 80             fields: "Items.AV,Items.WorkerStatuses"
 81         };
 82         // loop until item scanned or timedout
 83         (function loop(self) {
 84             var now = new Date().getTime();
 85             if(now >= then) {
 86                 complete("timedout");
 87                 return;
 88             }
 89             self.search.searchAll(ticket, function(jqXHR, textStatus) {
 90                 if(jqXHR.status != 200) {
 91                     complete("error");
 92                     return;
 93                 }
 94                 var errors = 0;
 95                 var pending = 0;
 96                 var items = JSON.parse(jqXHR.responseText).Items;
 97                 if(items) {
 98                     var av = items[0];
 99                     var ws = items[0].WorkerStatuses;
100                     for(var i = 0; i < workers.length; i++) {
101                         var status = "";
102                         if(workers[i] == "AV")
103                             status = _getWorkerStatus(av, "AV");
104                         else
105                             status = _getWorkerStatus(ws, workers[i]);
106                         console.log("worker: " + workers[i] +
107                                       ", status: " + status);
108                         if(status == "pending")
109                             pending++;
110                         else if(status == "error")
111                             errors++;
112                     }
113                 }
114                 if(0 !== errors) {
115                     // worker has failed
116                     complete("error");
117                 } else if(0 === pending) {
118                     // all workers done
119                     complete("success");
120                 } else {
121                     // try again after one second
122                     setTimeout(loop, 1000, self);
123                 }
124             }, params);
125         })(this);
126     };
127 
128     // Return the constructor function
129     return Fsio;
130 });
